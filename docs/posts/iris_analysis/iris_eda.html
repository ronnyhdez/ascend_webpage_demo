iris Analysis

Iris

This is my iris EDA

Library

library(dplyr)
library(ggplot2)
library(janitor)
library(rmarkdown)
library(plotly)

Data

iris_clean <- iris %>% 
  clean_names()

iris_clean %>% 
  paged_table()

Visualization

iris_plot <- iris_clean %>% 
  ggplot(aes(x = petal_length, y = petal_width, color = species)) +
  geom_point() +
  scale_color_viridis_d() +
  theme_light(base_size = 16)

iris_plot

Interactive plot

ggplotly(iris_plot)
